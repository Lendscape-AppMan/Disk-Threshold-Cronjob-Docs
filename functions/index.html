    def exec_df_on_node(node_name: str, volume_path: str) -> Optional[str]:
    """
    Execute 'df' command on a specified node using the Kubernetes Python client.

    Args:
        node_name (str): The name of the node where the command should be executed.
        volume_path (str): The path to the volume on the node for which the 'df' command will be executed.

    Returns:
        Optional[str]: The output of the 'df' command if successful, None if an error occurs during execution.

    Raises:
        ApiException: If no suitable pod is found on the specified node.
    """
    config.load_incluster_config()
    api_instance = client.CoreV1Api()

    # Find the right pod on the node
    pod_list = api_instance.list_pod_for_all_namespaces(
        field_selector=f"spec.nodeName={node_name}"
    )
    pod_name = next(
        (
            pod.metadata.name
            for pod in pod_list.items
            if pod.metadata.namespace == "kube-system"
        ),
        None,
    )
    if not pod_name:
        raise ValueError(f"No suitable pod found on node {node_name}")

    # Execute 'df' command in pod
    exec_command = ["/bin/sh", "-c", f"df -h {volume_path}"]
    try:
        return stream.stream(
            api_instance.connect_get_namespaced_pod_exec,
            pod_name,
            "kube-system",
            command=exec_command,
            stderr=True,
            stdin=False,
            stdout=True,
            tty=False,
        )
    except ApiException as e:
        print(f"Error executing 'df' command on node {node_name}: {str(e)}")
        return None

    def get_all_node_names() -> List[str]:
    """
    Returns a list of all node names in the Kubernetes cluster.

    Returns:
        List[str]: A list of all node names in the cluster.
    """
    config.load_incluster_config()
    api_instance = client.CoreV1Api()
    nodes = api_instance.list_node()
    return [node.metadata.name for node in nodes.items]

    def get_last_alert_timestamp(namespace: Optional[str] = "default") -> datetime:
    """
    Get the timestamp of the last alert from the disk-usage-alert-timestamp ConfigMap.

    Args:
        namespace (Optional[str]): The namespace where the ConfigMap is located. Set to "default" if none provided.

    Returns:
        datetime: The timestamp of the last alert as a datetime object.

    Raises:
        client.exceptions.ApiException: If the Kubernetes API call fails and the error is not a "Not Found" error.

    Examples:
        >>> get_last_alert_timestamp()
        datetime.datetime(2022, 3, 29, 17, 30, 0)
    """
    config.load_incluster_config()
    api_instance = client.CoreV1Api()

    try:
        config_map = api_instance.read_namespaced_config_map(
            name=CONFIGMAP_NAME, namespace=namespace
        )
        timestamp = config_map.data.get("timestamp")
        return (
            datetime.fromtimestamp(int(timestamp))
            if timestamp
            else datetime.fromtimestamp(0)
        )
    except client.exceptions.ApiException as e:
        if e.status == 404:
            return datetime.fromtimestamp(0)
        else:
            raise e

    def main():
    """
    Check disk usage on multiple Kubernetes nodes and send an alert to MS Teams if disk usage exceeds a threshold.

    First, retrieves the last alert timestamp from a Kubernetes ConfigMap using `get_last_alert_timestamp()`.

    If the difference between the current time and the last alert timestamp is less than the alert interval, the function exits.

    Otherwise, it iterates over the specified `NODES` and retrieves the disk usage for a mounted volume using `subprocess.check_output()`.

    If the disk usage exceeds the `THRESHOLD`, it sends an alert to Teams using `send_teams_alert()`.

    If an alert is sent, the function updates the last alert timestamp in the ConfigMap using `update_last_alert_timestamp()`.

    Example:
        >>> main()
        Alert: Disk usage on node hpd-elasticsearch-es-data-hot-1 is at 85.6%, which is above the desired 70% threshold.
    """
    last_alert_timestamp = get_last_alert_timestamp()
    current_time = datetime.now()

    if current_time - last_alert_timestamp < ALERT_INTERVAL:
        print("Last alert was sent less than 5 hours ago, exiting...")
        return

    alert_sent = False
    NODES = get_all_node_names()
    for node in NODES:
        df_output = exec_df_on_node(node, "/path/to/mounted/volume")
        if df_output is not None:
            disk_usage_line = df_output.splitlines()[1]
            disk_usage = int(disk_usage_line.split()[-2].rstrip("%"))

            if disk_usage > THRESHOLD:
                send_teams_alert(node, disk_usage)
                alert_sent = True
                break

    if alert_sent:
        update_last_alert_timestamp(current_time)

    def send_teams_alert(node: str, disk_usage: float) -> None:
    """
    Send an alert to Teams when the disk usage exceeds a threshold.

    Args:
        node (str): The name of the node where disk usage exceeded the threshold.
        disk_usage (float): The percentage of disk usage that triggered the alert.

    Raises:
        requests.exceptions.RequestException: If the HTTP request fails.
    """
    payload = {
        "@type": "MessageCard",
        "@context": "http://schema.org/extensions",
        "text": f"Alert: Disk usage on node {node} is at {disk_usage}%, which is above the desired {THRESHOLD}% threshold.",
    }
    requests.post(TEAMS_WEBHOOK_URL, json=payload)

    def update_last_alert_timestamp(
    timestamp: datetime, namespace: Optional[str] = "default"
) -> None:
    """
    Update timestamp of the last alert in ConfigMap.

    Args:
        timestamp (datetime): The new timestamp to update the ConfigMap with.
        namespace (Optional[str]): The namespace where the ConfigMap is located. Set to "default" if none provided.

    Raises:
        client.exceptions.ApiException: If the Kubernetes API call fails.
    """
    config.load_incluster_config()
    api_instance = client.CoreV1Api()

    # Check ConfigMap exists
    try:
        api_instance.read_namespaced_config_map(
            name=CONFIGMAP_NAME, namespace=namespace
        )
        configmap_exists = True
    except client.exceptions.ApiException as e:
        if e.status == 404:
            configmap_exists = False
        else:
            raise e

    # Update or create ConfigMap with the new timestamp
    new_configmap = client.V1ConfigMap(
        metadata=client.V1ObjectMeta(name=CONFIGMAP_NAME),
        data={"timestamp": str(int(timestamp.timestamp()))},
    )

    if configmap_exists:
        api_instance.patch_namespaced_config_map(
            name=CONFIGMAP_NAME, namespace=namespace, body=new_configmap
        )
    else:
        api_instance.create_namespaced_config_map(
            namespace=namespace, body=new_configmap
        )
